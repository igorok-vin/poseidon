1   package com.nnk.springboot.service;
2   
3   import com.nnk.springboot.domain.Rating;
4   import com.nnk.springboot.exception.RatingNotFoundException;
5   import com.nnk.springboot.repositories.RatingRepository;
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;
8   import org.springframework.beans.factory.annotation.Autowired;
9   import org.springframework.stereotype.Service;
10  
11  import java.util.List;
12  import java.util.Optional;
13  
14  @Service
15  public class RatingService {
16      private final Logger logger = LoggerFactory.getLogger(RatingService.class);
17      @Autowired
18      RatingRepository ratingRepository;
19  
20      /**
21       * Get a list of all rating
22       * @return list of Rating containing all ratings
23       */
24      public List<Rating> getAllRating() {
25          logger.info("Service: display rating list");
26          return ratingRepository.findAll();
27      }
28  
29      /**
30       * Save a new Rating
31       * @param rating
32       * @return Rating saved
33       */
34      public Rating saveRating (Rating rating) {
35          logger.info("Service: save new rating to DB");
36          return ratingRepository.save(rating);
37      }
38  
39      /**
40       * Get rating by id.
41       * @param id Rating that containing id of the Rating
42       * @return Instance of Rating
43       */
44      public Rating getRatingById(Integer id) {
45          Optional<Rating> rating = ratingRepository.findById(id);
46          if(rating.isPresent()){
47              logger.info("Service: Rating with Id "+id+" was found");
48              return rating.get();
49          }else {
50              throw new RatingNotFoundException("CurvePoint not found");
51          }
52      }
53  
54      /**
55       * Update existing rating
56       * @param rating instance
57       * @return the Rating updated
58       */
59      public Rating updateRating(Rating rating) {
60          Rating ratingForUpdate = ratingRepository.getById(rating.getId());
61          ratingForUpdate.setFitchRating(rating.getFitchRating());
62          ratingForUpdate.setMoodysRating(rating.getMoodysRating());
63          ratingForUpdate.setSandPRating(ratingForUpdate.getSandPRating());
64          logger.info("Service: Rating updated with ID: " + rating.getId());
65          return ratingRepository.save(ratingForUpdate);
66      }
67  
68      /**
69       * Delete a Rating by id
70       * @param id Integer that containing id of the Rating
71       */
72      public void deleteRating(Integer id) {
73          logger.info("Service: Rating deleted with ID: " + id);
74          ratingRepository.deleteById(id);
75      }
76  }
